리터럴에 의한 객체 생성
var rect = {
width: 12,
height: 4,
getArea: function(){
area = this.width * this.height;
return area;
},
getCircum: function(){
circum = (parseInt(this.width) + parseInt(this.height)) * 2;
return circum;
}
};
객체안에 선언되지 않은 속성이나 메소드를 필요시 언제든지 동적으로 추가 해서 사용 할 수 있다.
rect.width = 10;
